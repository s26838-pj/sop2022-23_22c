Napisz program wyświetlający tradycyjną choinkę o zadanej wysokości.
Wysokość niech będzie podana jako parametr wywołania programu. 

#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
    int wysokosc;
    int i, j;

    if (argc != 2) {
        printf("Podaj wysokość choinki jako parametr wywołania programu!\n");
        return 1;
    }

    wysokosc = atoi(argv[1]);

    if (wysokosc <= 0) {
        printf("Podaj dodatnią wartość wysokości choinki!\n");
        return 1;
    }
    
    for (i = 0; i < wysokosc; i++) {
        for (j = 0; j < wysokosc - i - 1; j++) {
            printf(" ");
        }

        for (j = 0; j < 2 * i + 1; j++) {
            printf("*");
        }

        printf("\n");
    }

    return 0;
}
